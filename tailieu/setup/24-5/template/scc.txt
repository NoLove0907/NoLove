#define all(v) (v).begin(), (v).end()
using pii = pair<int, int>;
using vi = vector<int>;
#define FOR(i, a, b) for (int (i) = a; (i) <= (b); i++)
void mx(int& a, int b) { if (b > a) a = b; }
void mi(int& a, int b) { if (b < a) a = b; }
#define si(x) (int)(x.size())
const int INF = 1e18;
const int MOD = 1e9 + 7;

const int MAXN = 1e4 + 5;
int g[MAXN];
vi adj[MAXN];
int n, m, a, b;

int timer = 1, scc_cnt = 0;
vi num, low;
vector<bool> in_stack, del;
stack<int> s;
void dfs(int v) {
    num[v] = low[v] = timer++;
    s.push(v);
    for (int u : adj[v]) {
        if (del[u]) continue;
        if (num[u] == -1) {
            dfs(u);
            mi(low[v], low[u]);
        } else mi(low[v], num[u]);
    }

    if (num[v] == low[v] && !s.empty()) {
        scc_cnt++;
        db(scc_cnt)
        int pop_item;
         do {
            pop_item = s.top(); s.pop();
            del[pop_item] = true;
            // to print all scc
            db(pop_item);
        } while (pop_item != v);
    }
}

        adj[a].pb(b);
    }

    num = low = vi(n + 1, -1);
    in_stack = del = vector<bool>(n + 1, 0);
   
    FOR(i, 1, n)
        if (num[i] == -1)
            dfs(i);

    cout << scc_cnt;
}
